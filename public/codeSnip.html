    class Shortcut{
        constructor(keys, description, created){
            this.keys = keys;
            this.description = description;
            this.created = created;
        }
        getKeys(){
            return this.keys
        }
        getDescription(){
            return this.description
        }
        getCreationDate(){
            return this.created.toLocaleDateString()
        }
    }
